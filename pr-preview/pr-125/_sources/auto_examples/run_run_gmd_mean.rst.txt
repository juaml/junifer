        "kind": "OasisVBMTestingDatagrabber",
    }

        {
            "name": "Schaefer1000x7_TrimMean80",

        },

            "name": "Schaefer1000x7_Mean",

        {
            "name": "Schaefer1000x7_Std",
            "kind": "ParcelAggregation",

    storage = {
        "kind": "SQLiteFeatureStorage",
    }

        run(

            datagrabber=datagrabber,
            markers=markers,
            storage=storage,
